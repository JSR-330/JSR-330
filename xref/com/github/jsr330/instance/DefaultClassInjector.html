42      protected Map<String, Provider<?>> providers = new TreeMap<String, Provider<?>>();
43      protected Map<String, Object> singletons = new TreeMap<String, Object>();
44      protected TypeDeterminator typeDeterminator = new DefaultTypeDeterminator();
45      protected Map<String, TypeContainer> types = new TreeMap<String, TypeContainer>();

47      @Override
48      public void injectStaticMembers(Map<String, Class<?>> classes, Map<String, Class<?>[]> inheritanceTree, ClassLoader classLoader) {
49          TypeContainer typeContainer;
50          
51          for (Map.Entry<String, Class<?>> type : classes.entrySet()) {
52              typeContainer = generateTypeContainer(type.getValue(), inheritanceTree, classLoader);
53              for (InjectionSet set : typeContainer.injectionSets) {
54                  injectStaticFields(set, null, inheritanceTree, classLoader);
55                  injectStaticMethods(set, null, inheritanceTree, classLoader);
56              }
57          }
58      }

60      @Override
61      @SuppressWarnings("unchecked")
62      public <T> T instance(Class<?> type, Map<String, Class<?>[]> inheritanceTree, ClassLoader classLoader, Class<?>[] generics, Annotation qualifier) {
63          T inst = null;
64          Provider<?> provider;
65          Class<?>[] candidates;
66          TypeContainer typeContainer;
67          
68          LOGGER.debug("instance - gets instance for {} with generics {}", type, generics);
69          
70          if (type.isAssignableFrom(Provider.class)) {
71              provider = new SimpleProvider(generics[0], this, inheritanceTree, qualifier, classLoader);
72              LOGGER.debug("instance - return provider {} for {}", provider, generics[0]);
73              return (T) provider;
74          }
75          
76          candidates = inheritanceTree.get(type.getName());
77          type = typeDeterminator.determineClass(type, candidates, qualifier, classLoader);
78          LOGGER.debug("instance - got type {} due to annotations", type);
79          
80          if (singletons.containsKey(type.getName())) {
81              LOGGER.debug("instance - deliver singleton {}", singletons.get(type.getName()));
82              return (T) singletons.get(type.getName());
83          } else {
84              if (type.isInterface() || Modifier.isAbstract(type.getModifiers())) {
85                  candidates = inheritanceTree.get(type.getName());
86                  if (candidates != null && candidates.length > 0) {
87                      if (candidates.length == 1) {
88                          inst = instance(candidates[0], inheritanceTree, classLoader, null, qualifier);
89                      }
90                  }
91                  
92                  if (inst == null) {
93                      inst = instance(type, inheritanceTree, classLoader, null, qualifier);
94                  }
95              } else {
96                  if ((typeContainer = types.get(type.getName())) == null) {
97                      typeContainer = generateTypeContainer(type, inheritanceTree, classLoader);
98                  }
99                  
100                 if (typeContainer != null && typeContainer.constructor != null) {
101                     try {
102                         inst = (T) typeContainer.constructor.newInstance(getArguments(typeContainer.constructor, inheritanceTree, classLoader));
103                         injectTypeContainer(typeContainer, inst, inheritanceTree, classLoader);
104                         
105                         if (type.isAnnotationPresent(Singleton.class) && !singletons.containsKey(type.getName())) {
106                             singletons.put(type.getName(), inst);
107                         }
108                     } catch (Exception exception) {
109                         LOGGER.debug("error while instancing type", exception);
110                     }
111                 }
112             }
113         }
114         
115         return inst;
116     }

118     protected TypeContainer generateTypeContainer(Class<?> type, Map<String, Class<?>[]> inheritanceTree, ClassLoader classLoader) {
119         Constructor<?> ctor;
120         TypeContainer typeContainer = null;
121         
122         ctor = getInjectableConstructor(type);
123         if (ctor == null) {
124             ctor = getDefaultConstructor(type);
125         }
126         
127         typeContainer = new TypeContainer(type, ctor);
128         typeContainer.gatherInformation();
129         types.put(type.getName(), typeContainer);
130         
131         return typeContainer;
132     }

134     protected void injectTypeContainer(TypeContainer typeContainer, Object inst, Map<String, Class<?>[]> inheritanceTree, ClassLoader classLoader) {
135         for (InjectionSet set : typeContainer.injectionSets) {
136             injectFields(set, inst, inheritanceTree, classLoader);
137             injectMethods(set, inst, inheritanceTree, classLoader);
138         }
139     }

149     protected void injectFields(InjectionSet set, Object inst, Map<String, Class<?>[]> inheritanceTree, ClassLoader classLoader, boolean onlyStatic) {
150         ParameterizedType parameterizedType;
151         Class<?> objectType;
152         Class<?>[] generics;
153         Class<?>[] candidates;
154         Annotation qualifier;
155         
156         for (Field field : onlyStatic ? set.staticFields : set.fields) {
157             try {
158                 LOGGER.debug("injectFields - field injected {} of {}", field, set.type);
159                 if (field.getGenericType() instanceof ParameterizedType) {
160                     parameterizedType = (ParameterizedType) field.getGenericType();
161                     generics = getGenericTypes(parameterizedType);
162                     objectType = null;
163                     if (parameterizedType.getRawType() instanceof Class) {
164                         objectType = (Class<?>) parameterizedType.getRawType();
165                     }
166                     
167                     if (objectType != null) {
168                         candidates = inheritanceTree.get(objectType.getName());
169                         qualifier = getQualifier(field.getAnnotations());
170                         objectType = typeDeterminator.determineClass(objectType, candidates, qualifier, classLoader);
171                         field.set(inst, instance(objectType, inheritanceTree, classLoader, generics, qualifier));
172                     }
173                 } else if (field.getGenericType() instanceof Class) {
174                     qualifier = getQualifier(field.getAnnotations());
175                     field.set(inst, instance((Class<?>) field.getGenericType(), inheritanceTree, classLoader, null, qualifier));
176                 }
177             } catch (Exception exception) {
178                 LOGGER.debug("Error while injecting field", exception);
179             }
180         }
181     }

191     protected void injectMethods(InjectionSet set, Object inst, Map<String, Class<?>[]> inheritanceTree, ClassLoader classLoader, boolean onlyStatic) {
192         Object[] arguments;
193         
194         for (Method method : onlyStatic ? set.staticMethods : set.methods) {
195             try {
196                 LOGGER.debug("injectMethods - method injected {} of {}", method, set.type);
197                 arguments = getArguments(method, inheritanceTree, classLoader);
198                 if (arguments != null) {
199                     method.invoke(inst, arguments);
200                 } else {
201                     method.invoke(inst);
202                 }
203             } catch (Exception exception) {
204                 LOGGER.debug("error while invoking method", exception);
205             }
206         }
207     }

217     protected Object[] getArguments(Map<String, Class<?>[]> inheritanceTree, Class<?>[] parameters, Type[] generics, Annotation[][] annotations,
218             ClassLoader classLoader) {
219         Object[] arguments = null;
220         Class<?>[] genericTypes;
221         Class<?>[] candidates;
222         int index = 0;
223         Class<?> cls;
224         Annotation qualifier;
225         
226         if (parameters.length > 0) {
227             arguments = new Object[parameters.length];
228             for (Class<?> parameterType : parameters) {
229                 genericTypes = null;
230                 if (ParameterizedType.class.isAssignableFrom(generics[index].getClass())) {
231                     genericTypes = getGenericTypes((ParameterizedType) generics[index]);
232                 }
233                 
234                 candidates = inheritanceTree.get(parameterType.getName());
235                 qualifier = getQualifier(annotations[index]);
236                 cls = typeDeterminator.determineClass(parameterType, candidates, qualifier, classLoader);
237                 arguments[index] = instance(cls, inheritanceTree, classLoader, genericTypes, qualifier);
238                 index++;
239             }
240         }
241         
242         return arguments;
243     }

245     protected Annotation getQualifier(Annotation[] annotations) {
246         if (annotations != null && annotations.length > 0) {
247             for (Annotation annotation : annotations) {
248                 if (annotation.annotationType().isAnnotationPresent(Qualifier.class)) {
249                     return annotation;
250                 }
251             }
252         }
253         
254         return null;
255     }
256     
257     protected Constructor<?> getInjectableConstructor(Class<?> type) {
258         Constructor<?>[] ctors = type.getDeclaredConstructors();
259         
260         for (Constructor<?> ctor : ctors) {
261             if (ctor.isAnnotationPresent(Inject.class)) {
262                 LOGGER.debug("getInjectableConstructor - constructor found for {} {}", type, ctor);
263                 if (!ctor.isAccessible()) {
264                     ctor.setAccessible(true);
265                 }
266                 return ctor;
267             }
268         }
269         
270         return null;
271     }
272     
273     protected Constructor<?> getDefaultConstructor(Class<?> type) {
274         Constructor<?>[] ctors = type.getDeclaredConstructors();
275         
276         for (Constructor<?> ctor : ctors) {
277             if (ctor.getParameterTypes().length == 0) {
278                 if (!ctor.isAccessible()) {
279                     ctor.setAccessible(true);
280                 }
281                 LOGGER.debug("getDefaultConstructor - got constructor for {}", type);
282                 return ctor;
283             }
284         }
285         
286         return null;
287     }

289     protected Class<?>[] getGenericTypes(ParameterizedType type) {
290         List<Class<?>> list = new ArrayList<Class<?>>();
291         
292         for (Type tmp : type.getActualTypeArguments()) {
293             if (tmp instanceof Class) {
294                 list.add((Class<?>) tmp);
295             }
296         }
297         
298         return list.toArray(new Class<?>[] {});
299     }
