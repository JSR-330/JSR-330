16  package com.github.jsr330.spi;
17  
18  import java.lang.annotation.Annotation;
19  import java.util.Map;
20  
21  import com.github.jsr330.Injector;
22  
23  /**
24   * A class injector does the actual work for the DI framework: instance all static methods and fields once and getting an instance of type X
25   * with respect to the possibly assigned {@link TypeConfig}.
26   */
27  public interface ClassInjector {
28      
29      /**
30       * Injects all static fields and methods of the classes specified once in a lifetime of an {@link Injector}.
31       * 
32       * @param classes The classes which static members are going to be injected.
33       * @param inheritanceTree The entire inheritance tree of all known classes.
34       * @param classLoader The class loader which loaded all the classes.
35       */
36      void injectStaticMembers(Map<String, Class<?>> classes, Map<String, Class<?>[]> inheritanceTree, ClassLoader classLoader);
37      
38      /**
39       * Get an injected instance of the specified type back.
40       * 
41       * @param type The type to be instanced.
42       * @param generics The type parameters of a may generic type - can be {@code null}.
43       * @param qualifier A possible qualifier for the instance of the type.
44       * @param inheritanceTree The entire inheritance tree of all known classes.
45       * @param classLoader The class loader which loaded all the classes.
46       * 
47       * @return Returns the injected instance of the specified type.
48       */
49      <T> T instance(Class<T> type, Map<String, Class<? extends T>[]> inheritanceTree, ClassLoader classLoader, Class<?>[] generics, Annotation qualifier);
50      
51      /**
52       * Assigns a new {@link TypeConfig}.
53       * 
54       * @param config The new TypeConfig - can be {@code null}.
55       */
56      void setTypeConfig(TypeConfig config);
57      
58  }
